Discord Bot Integration

Use the following boilerplate code to create a Discord bot that lets members register with their name and email.

Your club API endpoint: https://{{ request.get_host }}/api/v1/clubs/{{ club.slug }}/members/

You will need an API token. Generate one for a club admin account in the Django admin.

import discord
from discord import app_commands
import aiohttp

API_URL = "https://{{ request.get_host }}/api/v1/clubs/{{ club.slug }}/members/"
API_TOKEN = "YOUR_API_TOKEN_HERE"  # Set this to your admin user's token

class RegistrationBot(discord.Client):
    def __init__(self):
        super().__init__(intents=discord.Intents.default())
        self.tree = app_commands.CommandTree(self)

    async def setup_hook(self):
        await self.tree.sync()

client = RegistrationBot()

@client.tree.command(name="register", description="Register your membership")
@app_commands.describe(
    first_name="Your first name",
    last_name="Your last name",
    email="Your email address"
)
async def register(interaction: discord.Interaction, first_name: str, last_name: str, email: str):
    """Register as a club member via Discord"""
    await interaction.response.defer(ephemeral=True)

    payload = {
        "first_name": first_name,
        "last_name": last_name,
        "email": email,
        "discord_id": str(interaction.user.id),
        "source": "discord"
    }
    headers = {"Authorization": f"Token {API_TOKEN}"}

    async with aiohttp.ClientSession() as session:
        async with session.post(API_URL, json=payload, headers=headers) as resp:
            if resp.status == 201:
                await interaction.followup.send(
                    f"✅ Welcome, {first_name}! You've been registered as a member.",
                    ephemeral=True
                )
            elif resp.status == 400:
                data = await resp.json()
                await interaction.followup.send(f"❌ Registration failed: {data}", ephemeral=True)
            else:
                await interaction.followup.send("❌ An error occurred. Please try again later.", ephemeral=True)

client.run("YOUR_BOT_TOKEN_HERE")

Save this as bot.py and install dependencies: pip install discord.py aiohttp
